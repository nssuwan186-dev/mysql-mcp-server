```python
async def handle_callback(update: Update, context: ContextTypes.DEFAULT_TYPE):
    """‡∏à‡∏±‡∏î‡∏Å‡∏≤‡∏£ Callback ‡∏à‡∏≤‡∏Å‡∏õ‡∏∏‡πà‡∏° Inline Keyboard"""
    query = update.callback_query
    await query.answer()
    
    action = query.data
    user_id = update.effective_user.id
    
    # ‡∏™‡πà‡∏á action ‡πÑ‡∏õ‡∏¢‡∏±‡∏á AI Agent
    response = await ai_agent.handle_action(
        action=action,
        user_id=user_id,
        context=context.user_data
    )
    
    # ‡πÅ‡∏Å‡πâ‡πÑ‡∏Ç‡∏Ç‡πâ‡∏≠‡∏Ñ‡∏ß‡∏≤‡∏°‡πÄ‡∏î‡∏¥‡∏°
    if response.get('keyboard'):
        reply_markup = InlineKeyboardMarkup(response['keyboard'])
        await query.edit_message_text(
            text=response['text'],
            reply_markup=reply_markup,
            parse_mode='HTML'
        )
    else:
        await query.edit_message_text(
            text=response['text'],
            parse_mode='HTML'
        )

async def handle_document(update: Update, context: ContextTypes.DEFAULT_TYPE):
    """‡∏à‡∏±‡∏î‡∏Å‡∏≤‡∏£‡πÑ‡∏ü‡∏•‡πå‡∏ó‡∏µ‡πà‡∏™‡πà‡∏á‡∏°‡∏≤ (Excel, PDF, etc.)"""
    document = update.message.document
    file = await document.get_file()
    
    # Download file
    file_path = f"uploads/{document.file_name}"
    await file.download_to_drive(file_path)
    
    await update.message.reply_text(
        f"‚úÖ ‡πÑ‡∏î‡πâ‡∏£‡∏±‡∏ö‡πÑ‡∏ü‡∏•‡πå {document.file_name} ‡πÅ‡∏•‡πâ‡∏ß\n"
        "‡∏Å‡∏≥‡∏•‡∏±‡∏á‡∏õ‡∏£‡∏∞‡∏°‡∏ß‡∏•‡∏ú‡∏•..."
    )
    
    # ‡∏õ‡∏£‡∏∞‡∏°‡∏ß‡∏•‡∏ú‡∏•‡πÑ‡∏ü‡∏•‡πå‡∏î‡πâ‡∏ß‡∏¢ AI Agent
    response = await ai_agent.process_file(
        file_path=file_path,
        file_type=document.mime_type,
        user_id=update.effective_user.id
    )
    
    await update.message.reply_text(response['text'], parse_mode='HTML')

def main():
    """‡πÄ‡∏£‡∏¥‡πà‡∏°‡∏ï‡πâ‡∏ô Bot"""
    # ‡∏™‡∏£‡πâ‡∏≤‡∏á Application
    application = Application.builder().token(settings.TELEGRAM_BOT_TOKEN).build()
```

```python
import anthropic
import json
from typing import Dict, Any
from config import settings
from .tools import TOOLS, execute_tool

class HotelAIAgent:
    def __init__(self):
        self.client = anthropic.Anthropic(api_key=settings.ANTHROPIC_API_KEY)
        self.model = "claude-sonnet-4-20250514"
        
        # System Prompt
        self.system_prompt = """
‡∏Ñ‡∏∏‡∏ì‡∏Ñ‡∏∑‡∏≠ AI Assistant ‡∏™‡∏≥‡∏´‡∏£‡∏±‡∏ö‡∏£‡∏∞‡∏ö‡∏ö‡∏à‡∏±‡∏î‡∏Å‡∏≤‡∏£‡πÇ‡∏£‡∏á‡πÅ‡∏£‡∏°
‡∏Ñ‡∏∏‡∏ì‡∏ä‡πà‡∏ß‡∏¢‡∏ú‡∏π‡πâ‡∏à‡∏±‡∏î‡∏Å‡∏≤‡∏£‡πÅ‡∏•‡∏∞‡∏û‡∏ô‡∏±‡∏Å‡∏á‡∏≤‡∏ô‡πÉ‡∏ô‡∏Å‡∏≤‡∏£:
- ‡∏à‡∏±‡∏î‡∏Å‡∏≤‡∏£‡∏´‡πâ‡∏≠‡∏á‡∏û‡∏±‡∏Å‡πÅ‡∏•‡∏∞‡∏Å‡∏≤‡∏£‡∏à‡∏≠‡∏á
- ‡∏ö‡∏±‡∏ô‡∏ó‡∏∂‡∏Å‡∏£‡∏≤‡∏¢‡∏£‡∏±‡∏ö-‡∏£‡∏≤‡∏¢‡∏à‡πà‡∏≤‡∏¢
- ‡∏Ñ‡πâ‡∏ô‡∏´‡∏≤‡∏Ç‡πâ‡∏≠‡∏°‡∏π‡∏•‡∏•‡∏π‡∏Å‡∏Ñ‡πâ‡∏≤
- ‡∏™‡∏£‡πâ‡∏≤‡∏á‡∏£‡∏≤‡∏¢‡∏á‡∏≤‡∏ô‡∏Å‡∏≤‡∏£‡πÄ‡∏á‡∏¥‡∏ô
- ‡∏à‡∏±‡∏î‡∏Å‡∏≤‡∏£‡∏û‡∏ô‡∏±‡∏Å‡∏á‡∏≤‡∏ô‡πÅ‡∏•‡∏∞‡πÄ‡∏á‡∏¥‡∏ô‡πÄ‡∏î‡∏∑‡∏≠‡∏ô
- ‡∏ï‡∏≠‡∏ö‡∏Ñ‡∏≥‡∏ñ‡∏≤‡∏°‡πÄ‡∏Å‡∏µ‡πà‡∏¢‡∏ß‡∏Å‡∏±‡∏ö‡∏Å‡∏≤‡∏£‡∏î‡∏≥‡πÄ‡∏ô‡∏¥‡∏ô‡∏á‡∏≤‡∏ô

‡∏ï‡∏≠‡∏ö‡∏î‡πâ‡∏ß‡∏¢‡∏†‡∏≤‡∏©‡∏≤‡πÑ‡∏ó‡∏¢‡∏ó‡∏µ‡πà‡∏™‡∏∏‡∏†‡∏≤‡∏û‡πÅ‡∏•‡∏∞‡∏Å‡∏£‡∏∞‡∏ä‡∏±‡∏ö
‡πÉ‡∏ä‡πâ emoji ‡πÄ‡∏û‡∏∑‡πà‡∏≠‡πÉ‡∏´‡πâ‡∏ô‡πà‡∏≤‡∏™‡∏ô‡πÉ‡∏à
‡∏ñ‡∏≤‡∏°‡∏ä‡∏±‡∏î‡πÄ‡∏à‡∏ô‡πÄ‡∏°‡∏∑‡πà‡∏≠‡∏ï‡πâ‡∏≠‡∏á‡∏Å‡∏≤‡∏£‡∏Ç‡πâ‡∏≠‡∏°‡∏π‡∏•‡πÄ‡∏û‡∏¥‡πà‡∏°‡πÄ‡∏ï‡∏¥‡∏°
"""
        
        # Conversation History (‡πÉ‡∏ä‡πâ Redis ‡πÉ‡∏ô Production)
        self.conversations = {}
    
    async def process_message(
        self, 
        message: str, 
        user_id: int,
        context: Dict = None
    ) -> Dict[str, Any]:
        """‡∏õ‡∏£‡∏∞‡∏°‡∏ß‡∏•‡∏ú‡∏•‡∏Ç‡πâ‡∏≠‡∏Ñ‡∏ß‡∏≤‡∏°‡∏à‡∏≤‡∏Å‡∏ú‡∏π‡πâ‡πÉ‡∏ä‡πâ"""
        
        # ‡∏î‡∏∂‡∏á conversation history
        if user_id not in self.conversations:
            self.conversations[user_id] = []
        
        conversation = self.conversations[user_id]
        conversation.append({
            "role": "user",
            "content": message
        })
        
        try:
            # ‡πÄ‡∏£‡∏µ‡∏¢‡∏Å Claude API
            response = self.client.messages.create(
                model=self.model,
                max_tokens=4096,
                system=self.system_prompt,
                messages=conversation,
                tools=TOOLS
            )
            
            # ‡∏ï‡∏£‡∏ß‡∏à‡∏™‡∏≠‡∏ö‡∏ß‡πà‡∏≤ Claude ‡∏ï‡πâ‡∏≠‡∏á‡∏Å‡∏≤‡∏£‡πÉ‡∏ä‡πâ Tool ‡∏´‡∏£‡∏∑‡∏≠‡πÑ‡∏°‡πà
            if response.stop_reason == "tool_use":
                # Execute tools
                tool_results = []
                assistant_message = {"role": "assistant", "content": response.content}
                conversation.append(assistant_message)
                
                for content_block in response.content:
                    if content_block.type == "tool_use":
                        tool_name = content_block.name
                        tool_input = content_block.input
                        
                        print(f"üîß Executing tool: {tool_name}")
                        print(f"   Input: {tool_input}")
                        
                        # Execute tool
                        tool_result = await execute_tool(tool_name, tool_input)
                        
                        tool_results.append({
                            "type": "tool_result",
                            "tool_use_id": content_block.id,
                            "content": json.dumps(tool_result, ensure_ascii=False)
                        })
                
                # ‡∏™‡πà‡∏á tool results ‡∏Å‡∏•‡∏±‡∏ö‡πÑ‡∏õ Claude
                conversation.append({
                    "role": "user",
                    "content": tool_results
                })
                
                # ‡πÄ‡∏£‡∏µ‡∏¢‡∏Å Claude ‡∏≠‡∏µ‡∏Å‡∏Ñ‡∏£‡∏±‡πâ‡∏á‡πÄ‡∏û‡∏∑‡πà‡∏≠‡πÉ‡∏´‡πâ‡∏ï‡∏≠‡∏ö
                final_response = self.client.messages.create(
                    model=self.model,
                    max_tokens=4096,
                    system=self.system_prompt,
                    messages=conversation,
                    tools=TOOLS
                )
                
                response_text = self._extract_text(final_response.content)
                
            else:
                response_text = self._extract_text(response.content)
            
            # ‡∏ö‡∏±‡∏ô‡∏ó‡∏∂‡∏Å assistant response
            conversation.append({
                "role": "assistant",
                "content": response_text
            })
            
            # ‡∏à‡∏≥‡∏Å‡∏±‡∏î‡∏Ñ‡∏ß‡∏≤‡∏°‡∏¢‡∏≤‡∏ß conversation (‡πÄ‡∏Å‡πá‡∏ö 20 ‡∏Ç‡πâ‡∏≠‡∏Ñ‡∏ß‡∏≤‡∏°‡∏•‡πà‡∏≤‡∏™‡∏∏‡∏î)
            if len(conversation) > 20:
                conversation = conversation[-20:]
            
            self.conversations[user_id] = conversation
            
            return {
                "text": response_text,
                "keyboard": self._extract_keyboard(response_text)
            }
            
        except Exception as e:
            print(f"Error in AI Agent: {e}")
            return {
                "text": f"‚ùå ‡πÄ‡∏Å‡∏¥‡∏î‡∏Ç‡πâ‡∏≠‡∏ú‡∏¥‡∏î‡∏û‡∏•‡∏≤‡∏î: {str(e)}"
            }

        action_messages = {

    def _extract_text(self, content) -> str:
        """‡πÅ‡∏¢‡∏Å‡∏Ç‡πâ‡∏≠‡∏Ñ‡∏ß‡∏≤‡∏°‡∏à‡∏≤‡∏Å response"""
        if isinstance(content, list):
            text_blocks = [block.text for block in content if hasattr(block, 'text')]
            return "\n".join(text_blocks)
        return str(content)
    
    def _extract_keyboard(self, text: str):
        """‡∏™‡∏£‡πâ‡∏≤‡∏á inline keyboard ‡∏à‡∏≤‡∏Å‡∏Ç‡πâ‡∏≠‡∏Ñ‡∏ß‡∏≤‡∏° (‡∏ñ‡πâ‡∏≤‡∏°‡∏µ)"""
        # Logic ‡∏™‡∏≥‡∏´‡∏£‡∏±‡∏ö‡∏™‡∏£‡πâ‡∏≤‡∏á keyboard ‡∏ï‡∏≤‡∏°‡∏ö‡∏£‡∏¥‡∏ö‡∏ó

        return None
```

```python
            target_date = today - timedelta(days=i)
            
            total = db.query(Room).count()
            occupied = db.query(Booking).filter(
                Booking.check_in_date <= target_date,
                Booking.check_out_date > target_date,
                Booking.status.in_(['confirmed', 'checked_in'])
            ).count()
            
            trend_data.append({
                "date": target_date.isoformat(),
                "occupancy_rate": round(occupied / total * 100, 1) if total > 0 else 0
            })
        
        # ‡∏Ñ‡∏≥‡∏ô‡∏ß‡∏ì‡πÅ‡∏ô‡∏ß‡πÇ‡∏ô‡πâ‡∏°
        avg_occupancy = sum(d['occupancy_rate'] for d in trend_data) / len(trend_data)
        recent_avg = sum(d['occupancy_rate'] for d in trend_data[-7:]) / 7
        trend = "‡πÄ‡∏û‡∏¥‡πà‡∏°‡∏Ç‡∏∂‡πâ‡∏ô" if recent_avg > avg_occupancy else "‡∏•‡∏î‡∏•‡∏á"
        
        return {
            "metric": "occupancy_trend",
            "period": f"{days_back} ‡∏ß‡∏±‡∏ô‡∏•‡πà‡∏≤‡∏™‡∏∏‡∏î",
            "data": trend_data,
            "analysis": {
                "average": round(avg_occupancy, 1),
                "recent_average": round(recent_avg, 1),
                "trend": trend,
                "change_percent": round(((recent_avg - avg_occupancy) / avg_occupancy * 100), 1)
            }
        }
    
    elif metric == "revenue_forecast":
        # ‡∏û‡∏¢‡∏≤‡∏Å‡∏£‡∏ì‡πå‡∏£‡∏≤‡∏¢‡πÑ‡∏î‡πâ (Simple Moving Average)
        from sqlalchemy import func
        
        # ‡∏î‡∏∂‡∏á‡∏£‡∏≤‡∏¢‡πÑ‡∏î‡πâ 30 ‡∏ß‡∏±‡∏ô‡∏•‡πà‡∏≤‡∏™‡∏∏‡∏î
        transactions = db.query(
            Transaction.transaction_date,
            func.sum(Transaction.amount).label('daily_income')
        ).filter(
            Transaction.type == 'income',
            Transaction.transaction_date >= date.today() - timedelta(days=30)
        ).group_by(Transaction.transaction_date).all()
        
        # ‡∏Ñ‡∏≥‡∏ô‡∏ß‡∏ì‡∏Ñ‡πà‡∏≤‡πÄ‡∏â‡∏•‡∏µ‡πà‡∏¢
        avg_daily = sum(t.daily_income for t in transactions) / len(transactions) if transactions else 0
        
        # ‡∏û‡∏¢‡∏≤‡∏Å‡∏£‡∏ì‡πå 7 ‡∏ß‡∏±‡∏ô‡∏Ç‡πâ‡∏≤‡∏á‡∏´‡∏ô‡πâ‡∏≤
        forecast = []
        for i in range(1, 8):
            forecast_date = date.today() + timedelta(days=i)
            # Simple forecast (‡πÉ‡∏ô production ‡πÉ‡∏ä‡πâ model ‡∏ó‡∏µ‡πà‡∏ã‡∏±‡∏ö‡∏ã‡πâ‡∏≠‡∏ô‡∏Å‡∏ß‡πà‡∏≤)
            forecast.append({
                "date": forecast_date.isoformat(),
                "predicted_income": float(avg_daily),
                "confidence": "medium"
            })
        
        return {
            "metric": "revenue_forecast",
            "historical_average": float(avg_daily),
            "forecast": forecast,
            "note": "‡∏û‡∏¢‡∏≤‡∏Å‡∏£‡∏ì‡πå‡∏à‡∏≤‡∏Å‡∏Ñ‡πà‡∏≤‡πÄ‡∏â‡∏•‡∏µ‡πà‡∏¢ 30 ‡∏ß‡∏±‡∏ô‡∏•‡πà‡∏≤‡∏™‡∏∏‡∏î"
        }
    
    elif metric == "guest_segmentation":
        # ‡πÅ‡∏ö‡πà‡∏á‡∏Å‡∏•‡∏∏‡πà‡∏°‡∏•‡∏π‡∏Å‡∏Ñ‡πâ‡∏≤
        from sqlalchemy import func
        
        segments = db.query(
            Guest.guest_level,
            func.count(Guest.id).label('count'),
            func.sum(Guest.total_spent).label('total_revenue'),
            func.avg(Guest.total_stays).label('avg_stays')
        ).group_by(Guest.guest_level).all()
        
        result = []
        for seg in segments:
            result.append({
                "level": seg.guest_level,
                "count": seg.count,
                "total_revenue": float(seg.total_revenue or 0),
                "avg_stays": round(float(seg.avg_stays or 0), 1),
                "avg_spend_per_guest": float((seg.total_revenue or 0) / seg.count)
            })
        
        return {
            "metric": "guest_segmentation",
            "segments": result
        }
    
    return {"error": "Unknown metric"}
```

```python
def rate_limit(max_requests=10, window=60):
    """Limit requests per user"""
    def decorator(func):
        @wraps(func)
        async def wrapper(update, context):
            user_id = update.effective_user.id
            now = time.time()
            
            # Clean old requests
            user_requests[user_id] = [
                req_time for req_time in user_requests[user_id]
```
